# MiniWallet API Endpoints - Full URLs for Postman Testing
# Base URL: http://localhost:8080 (assuming default Spring Boot port)

# =============================================================================
# CUSTOMER ENDPOINTS
# =============================================================================

# 1. Create Customer
POST http://localhost:8080/api/customers

# 2. Get Customer by ID
GET http://localhost:8080/api/customers/{id}
   Example: http://localhost:8080/api/customers/1

# 3. Update Customer
PUT http://localhost:8080/api/customers/{id}
   Example: http://localhost:8080/api/customers/1

# 4. Delete Customer
DELETE http://localhost:8080/api/customers/{id}
   Example: http://localhost:8080/api/customers/1

# =============================================================================
# WALLET ENDPOINTS
# =============================================================================

# 5. Create Wallet for Customer
POST http://localhost:8080/api/wallets?customerId={customerId}
   Example: http://localhost:8080/api/wallets?customerId=1

# 6. Get Wallet by ID
GET http://localhost:8080/api/wallets/{id}
   Example: http://localhost:8080/api/wallets/1

# 7. Get Wallet by Customer ID
GET http://localhost:8080/api/wallets/by-customer/{customerId}
   Example: http://localhost:8080/api/wallets/by-customer/1

# 8. Deposit Money to Wallet
POST http://localhost:8080/api/wallets/{id}/deposit
   Example: http://localhost:8080/api/wallets/1/deposit

# 9. Withdraw Money from Wallet
POST http://localhost:8080/api/wallets/{id}/withdraw
   Example: http://localhost:8080/api/wallets/1/withdraw

# 10. Transfer Money Between Wallets
POST http://localhost:8080/api/wallets/transfer

# =============================================================================
# TRANSACTION ENDPOINTS
# =============================================================================

# 11. Get Transaction by ID
GET http://localhost:8080/api/transactions/{id}
   Example: http://localhost:8080/api/transactions/TXN-001

# 12. Get Transactions by Wallet ID (with optional query parameters)
GET http://localhost:8080/api/transactions/by-wallet/{walletId}
   Example: http://localhost:8080/api/transactions/by-wallet/1

# Optional Query Parameters for filtering:
# - page: Page number (default: 0)
# - size: Page size (default: 20)
# - status: Transaction status filter
# - type: Transaction type filter
# - start: Start date (ISO format)
# - end: End date (ISO format)

# Examples with query parameters:
# http://localhost:8080/api/transactions/by-wallet/1?page=0&size=10
# http://localhost:8080/api/transactions/by-wallet/1?status=COMPLETED
# http://localhost:8080/api/transactions/by-wallet/1?type=DEPOSIT
# http://localhost:8080/api/transactions/by-wallet/1?start=2024-01-01T00:00:00&end=2024-12-31T23:59:59

# 13. Get Transactions by Reference
GET http://localhost:8080/api/transactions/by-reference/{reference}
   Example: http://localhost:8080/api/transactions/by-reference/REF-001

# 14. Get Transactions by Transfer ID
GET http://localhost:8080/api/transactions/by-transfer/{transferId}
   Example: http://localhost:8080/api/transactions/by-transfer/TRANSFER-001

# 15. Get All Transactions (paginated)
GET http://localhost:8080/api/transactions
   Example: http://localhost:8080/api/transactions

# Optional Query Parameters for pagination:
# - page: Page number (default: 0)
# - size: Page size (default: 20)

# Examples with query parameters:
# http://localhost:8080/api/transactions?page=0&size=50
# http://localhost:8080/api/transactions?page=1&size=10
